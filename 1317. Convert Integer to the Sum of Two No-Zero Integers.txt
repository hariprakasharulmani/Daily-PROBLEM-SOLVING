class Solution {
    public int[] getNoZeroIntegers(int n) {
        for(int i=1;i<n;i++){
            int b = n-i;
            if(f(i) && f(b)){
                return new int[]{i,b};
            }
        }
        return new int[]{-1,-1};
    }
    public static boolean f(int n){
        while(n>0){
            if(n%10==0){
                return false;
            }
            n/=10;
        }
        return true;
    }
}

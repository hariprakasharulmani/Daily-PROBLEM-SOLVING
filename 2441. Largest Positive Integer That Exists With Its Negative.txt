class Solution {
    public int findMaxK(int[] nums) {
        List<Integer>list = new ArrayList<>();
        for(int i:nums) list.add(i);
        int max = Integer.MIN_VALUE;
        for(int i:list){
            if(i>0 && list.contains(-i)){
                max = Math.max(max,i);
            }
        }

        return max==Integer.MIN_VALUE?-1:max;
    }
}

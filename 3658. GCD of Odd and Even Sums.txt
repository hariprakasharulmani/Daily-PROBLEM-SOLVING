class Solution {
    public int gcdOfOddEvenSums(int n) {
        int sumEven = 0;
        int sumOdd = 0;
        for(int i=1;i<=n*2;i++){
            if(i%2==0) sumEven+=i;
            else sumOdd+=i;
        }
        return gcd(sumEven,sumOdd);

    }
    public static int gcd(int n,int m){
        int size =  Math.max(n,m);
        for(int i=size;i>=1;i--){
            if(n%i==0 && m%i==0) return i;
        }
        return -1;
    }
}

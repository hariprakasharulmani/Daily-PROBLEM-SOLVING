class Solution {
    public int alternateDigitSum(int n) {
        int k = 0;
        while(n!=0){
            k=k*10+n%10;
            n/=10;
        }
        n=k;
        int sum = 0;
        int i=1;
        while(n!=0){
            int rem = n%10;
            sum = sum +(1*i*rem);
            i=-(i);
            n/=10;
        }
        return sum;
    }
}

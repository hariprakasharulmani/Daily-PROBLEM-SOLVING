class Solution {
    public int minimumIndex(int[] nums, int target) {
        int min = Integer.MAX_VALUE;
        int n = nums.length;
        for(int i=0;i<n;i++){
            if(nums[i]>=target){
                min =Math.min(nums[i],min);
            }
        }
        for(int i=0;i<n;i++){
            if(nums[i]==min){
                return i;
            }
        }
        return -1;
    }
}

class Solution {
    public int minimumChairs(String s) {
        int count = 0;
        int max = Integer.MIN_VALUE;
        for(int i=0;i<s.length();i++){
            char c = s.charAt(i);
            if(c=='E'){
                count+=1;
                max = Math.max(count,max);
            }
            else{
                count-=1;
                max = Math.max(count,max);
            }
        }
        return max;
    }
}
